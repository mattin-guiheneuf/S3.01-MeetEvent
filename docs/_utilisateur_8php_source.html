244 $reco = $this->getRecommandation()->getSuggestion();
245
246 // Trier les événements par pourcentage par ordre décroissant

250
251 // Déterminer quels événements recommander
252 $evenementsARecommander = [];

254 $evenementId = $paire['evenement'];
255 $pourcentage = $paire['pourcentage'];
256
257 // Filtrer les événements avec une similarité >= 0.7

259 $evenementsARecommander[] = $evenementId;
260 }
261 }
262

264 if (count($evenementsARecommander) < 5) {
